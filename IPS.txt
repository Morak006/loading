#!/bin/bash
hostnamectl set-hostname isp; exec bash
	sudo tee /etc/network/interfaces > /dev/null <<EOF
auto eth0
iface eth0 inet dhcp

auto eth1
iface eth1 inet static
    address 172.16.4.1/28

auto eth2
iface eth2 inet static
    address 172.16.5.1/28
EOF
net.ipv4.ip_forvard = 1
systemctl restart networking
iptables -t nat -A POSTROUTING -s 172.16.4.0/28 -j eth0 -o MASQUERADE
iptables -t nat -A POSTROUTING -s 172.16.5.0/28 -o eth0 -j MASQUERADE
iptables -t nat -L
iptables-save /root/rules
crontab -e
1
@reboot > /sbin/iptables-restore < /root/rules
$? -eq 0
